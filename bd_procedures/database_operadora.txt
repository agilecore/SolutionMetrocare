create table log_acesso
(
   id_usuario                 int                       not null ,
   dt_log                     date                      not null    
);

create table log_plano
(
    id_log_plano              int                       not null auto_increment ,
    id_alteracao              int                       not null  ,
    id_usuario                int                       not null  ,
    dt_alteracao              date                      not null  ,  
    primary key               (id_log_plano)
);

create table log_contrato
(
    id_log_contrato           int                       not null auto_increment ,
    id_alteracao              int                       not null  ,
    id_usuario                int                       not null  ,
    dt_alteracao              date                      not null  ,
    primary key               (id_log_contrato)
);

create table log_beneficiario
(
    id_log_beneficiario       int                       not null auto_increment ,
    id_usuario                int                       not null  ,
    id_alteracao              int                       not null  ,
    dt_alteracao              date                      not null  ,
    primary key               (id_log_beneficiario)
);

create table log_dependente
(
    id_log_dependente         int                       not null auto_increment ,
    id_usuario                int                       not null  ,
    id_alteracao              int                       not null  ,
    primary key               (id_log_dependente)
);

create table log_credenciado
(
    id_log_credenciado        int                       not null auto_increment ,
    id_usuario                int                       not null  ,
    id_alteracao              int                       not null  ,
    primary key               (id_log_credenciado)
);

create table log_prestador
(
    id_log_prestador          int                       not null auto_increment ,
    id_usuario                int                       not null  ,
    id_alteracao              int                       not null  ,
    primary key               (id_log_prestador)
);

create table sessions
(
	sessionid                 varchar(80)               null      ,
	applicationname           varchar(255)              null      ,
	created                   date                      null      ,
	expires                   date                      null      ,
	lockdate                  date                      null      ,
	lockid                    int                       null      ,
	timeout                   int                       null      ,
	locked                    TINYINT(1)                null      ,
	sessionitems              text                      null      ,
	flags                     int                       null      ,
	primary key               (sessionid,applicationname)
);
